Simple ImageButton ControlTemplate

This code lets you create a simple button that uses images.

Simple ControlTemplate (put this in your Window, Page, or Application's <Resources /> section).

 1   <ControlTemplate x:Key="ImageButton" TargetType="{x:Type Button}">
 2     <ContentPresenter HorizontalAlignment="Stretch" VerticalAlignment="Stretch" Width="Auto" Opacity="0.8" x:Name="contentPresenter">
 3     </ContentPresenter>
 4     <ControlTemplate.Triggers>
 5 
 6       <Trigger Property="IsMouseOver" SourceName="contentPresenter" Value="True">
 7         <Setter Property="OpacityMask" TargetName="contentPresenter" Value="#FF000000"/>
 8         <Setter Property="Opacity" TargetName="contentPresenter" Value="1"/>
 9       </Trigger>
10       <Trigger Property="IsPressed"  Value="True">
11         <Setter Property="Opacity" TargetName="contentPresenter" Value="0.3"/>
12       </Trigger>
13       <Trigger Property="IsEnabled" Value="False">
14         <Setter Property="Opacity" TargetName="contentPresenter" Value="0.2"/>
15       </Trigger>
16     </ControlTemplate.Triggers>
17   </ControlTemplate>
18

Or if you want to make it look ever cooler (though, a slightly more resource hungry), try the animated flavor!

Animated ControlTemplate

 1   <ControlTemplate x:Key="ImageButton" TargetType="{x:Type Button}">
 2    <ControlTemplate.Resources>
 3     <Storyboard x:Key="MouseOver">
 4      <DoubleAnimationUsingKeyFrames BeginTime="00:00:00" Storyboard.TargetName="contentPresenter" Storyboard.TargetProperty="(UIElement.Opacity)">
 5       <SplineDoubleKeyFrame KeyTime="00:00:00.2000000" Value="1"/>
 6      </DoubleAnimationUsingKeyFrames>
 7     </Storyboard>
 8     <Storyboard x:Key="MouseOut">
 9      <DoubleAnimationUsingKeyFrames BeginTime="00:00:00" Storyboard.TargetName="contentPresenter" Storyboard.TargetProperty="(UIElement.Opacity)">
10       <SplineDoubleKeyFrame KeyTime="00:00:00.3000000" Value="0.7"/>
11      </DoubleAnimationUsingKeyFrames>
12     </Storyboard>
13     <Storyboard x:Key="PressedOn">
14      <DoubleAnimationUsingKeyFrames BeginTime="00:00:00" Storyboard.TargetName="contentPresenter" Storyboard.TargetProperty="(UIElement.Opacity)">
15       <SplineDoubleKeyFrame KeyTime="00:00:00.0500000" Value="0.3"/>
16      </DoubleAnimationUsingKeyFrames>
17     </Storyboard>
18     <Storyboard x:Key="PressedOff">
19      <DoubleAnimationUsingKeyFrames BeginTime="00:00:00" Storyboard.TargetName="contentPresenter" Storyboard.TargetProperty="(UIElement.Opacity)">
20       <SplineDoubleKeyFrame KeyTime="00:00:00.1000000" Value="0.7"/>
21      </DoubleAnimationUsingKeyFrames>
22     </Storyboard>
23    </ControlTemplate.Resources>
24    <ContentPresenter HorizontalAlignment="Stretch" VerticalAlignment="Stretch" Width="Auto" Opacity="0.7" x:Name="contentPresenter" Cursor="Hand">
25    </ContentPresenter>
26    <ControlTemplate.Triggers>
27     <Trigger Property="IsMouseOver" SourceName="contentPresenter" Value="True">
28      <Trigger.EnterActions>
29       <BeginStoryboard Storyboard="{StaticResource MouseOver}"/>
30      </Trigger.EnterActions>
31      <Trigger.ExitActions>
32       <BeginStoryboard x:Name="MouseOut_BeginStoryboard" Storyboard="{StaticResource MouseOut}"/>
33      </Trigger.ExitActions>
34     </Trigger>
35     <Trigger Property="IsPressed"  Value="True">
36      <Trigger.EnterActions>
37       <BeginStoryboard x:Name="PressedOn_BeginStoryboard" Storyboard="{StaticResource PressedOn}"/>
38      </Trigger.EnterActions>
39      <Trigger.ExitActions>
40       <BeginStoryboard x:Name="PressedOff_BeginStoryboard" Storyboard="{StaticResource PressedOff}"/>
41      </Trigger.ExitActions>
42     </Trigger>
43     <Trigger Property="IsEnabled" Value="False">
44      <Setter Property="Opacity" TargetName="contentPresenter" Value="0.2"/>
45     </Trigger>
46    </ControlTemplate.Triggers>
47   </ControlTemplate>

Usage

1 <Button Template="{DynamicResource ImageButton}" Height="48">
2     <Image Width="Auto" Height="Auto" Source="images\ok.png" />
3 </Button>

Enjoy!
